Implementation Example

Simple, intuitive API for building complex agent workflows

neural_flow.py 
from neural_flow import Agent, ParallelLayer, SequentialLayer

# Initialize the agent with context
agent = Agent(
    system_prompt="You are a helpful AI assistant...",
    personality="professional, friendly",
    tools=[search, calculator, code_executor]
)

# Define parallel processing layer
parallel = ParallelLayer(agents=[
    MiniAgent(role="researcher"),
    MiniAgent(role="analyzer"),
    MiniAgent(role="synthesizer")
])

# Define sequential post-processing
sequential = SequentialLayer(steps=[
    "combine_answers",
    "inject_personality",
    "format_output"
])

# Execute the workflow
response = agent.execute(
    user_message="Explain quantum computing",
    layers=[parallel, sequential]
)
